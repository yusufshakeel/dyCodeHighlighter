  • .dyCodeHighlighter
  • .line-numbers

  • data-dyCodeHighlighter-highlight="11,20,22"
  • data-dyCodeHighlighter-custom-style='{"backgroundColor": "#7b68ee", "color": "white", "highlightBgColor": "white", "highlightColor": "black", "thickLeftBorderColor": "#f4a460", "borderColor": "#f4a460", "fontSize": "24px", "lineNumbersBorderColor": "#fff"}'

custom style JSON

/**
 * This is a sample program to print the following pattern.
 *
 * H
 * Ha
 * Hap
 * Happ
 * Happy
 */
var
  str = "Happy",
  len = str.length,
  r,
  c,
  pattern;

for (r = 0; r < len; r++) {
  pattern = '';
  for (c = 0; c <= r; c++) {
    pattern += str[c];
  }
  console.log(pattern);
}
